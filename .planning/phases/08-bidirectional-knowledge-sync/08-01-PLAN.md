def content_hash(text: str) -> str  # SHA256[:16] of utf-8 encoded text
def post_to_mem0(text: str, metadata: dict) -> bool  # POST {text, agent_id, metadata}; timeout=120
def main()                        # entry point: glob journals, apply 3 guards, sync, save state

# State shape (stored under "obsidian-journals" key in state dict):
# {
#   "obsidian-journals": {
#     "last_mtime": "2026-04-12T..." | null,
#     "processed_hashes": ["abc123def456", ...]
#   }
# }

# main() behaviour summary:
# 1. Exits sys.exit(1) if JOURNALS_DIR does not exist
# 2. Loads state, extracts obs_state["processed_hashes"] and obs_state["last_mtime"]
# 3. For each ????-??-??.md in JOURNALS_DIR (sorted):
#    a. Read content (utf-8, errors=replace); skip if empty
#    b. Guard 2: if content_hash(content) in processed_hashes → skipped_hash++, continue
#    c. Guard 3: if mtime <= cutoff_ts → skipped_mtime++, continue
#    d. POST to mem0; if success: add hash to processed_hashes, update last_mtime
#    e. If POST fails: print warning, do NOT add hash or update mtime
# 4. Saves state (always, even if nothing ingested)
# 5. Prints summary line: "N synced, N duplicate, N unchanged"
