```python
"""
TDD suite for skill-sync.py .hermes-staging pickup (AGENT-02).

RED phase: run before implementing STAGING_DIR block — all 6 tests fail.
GREEN phase: run after implementation — all 6 pass.
"""
import importlib
import json
import shutil
import sys
from pathlib import Path
from unittest.mock import patch, MagicMock

import pytest

# Import hyphenated module via importlib
SCRIPTS_DIR = Path(__file__).parent.parent
if str(SCRIPTS_DIR) not in sys.path:
    sys.path.insert(0, str(SCRIPTS_DIR))

m = importlib.import_module("skill-sync")


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def make_skill(parent: Path, name: str, with_skill_md: bool = True) -> Path:
    """Create a minimal skill directory under parent."""
    skill_dir = parent / name
    skill_dir.mkdir(parents=True, exist_ok=True)
    if with_skill_md:
        (skill_dir / "SKILL.md").write_text(f"# {name}\nA test skill.")
    return skill_dir


def patch_config(tmp_path, monkeypatch):
    """
    Redirect CONFIG paths into tmp_path so tests are isolated.
    Returns (staging_dir, master_dir, contributions_log).
    """
    master_dir = tmp_path / "skills"
    master_dir.mkdir()
    staging_dir = master_dir / ".hermes-staging"
    staging_dir.mkdir()
    contributions_log = tmp_path / "skill-contributions.jsonl"

    new_config = dict(m.CONFIG)
    new_config["master_dir"] = master_dir
    new_config["state_file"] = tmp_path / "skill-sync-state.json"
    new_config["openclaw_dir"] = tmp_path / "openclaw-skills"
    new_config["openclaw_dir"].mkdir()
    new_config["hermes_dir"] = tmp_path / "hermes-skills"   # does not exist — skips that block

    monkeypatch.setattr(m, "CONFIG", new_config)
    monkeypatch.setattr(m, "STAGING_DIR", staging_dir)
    monkeypatch.setattr(m, "CONTRIBUTIONS_LOG", contributions_log)

    return staging_dir, master_dir, contributions_log


# ---------------------------------------------------------------------------
# Tests
# ---------------------------------------------------------------------------

def test_staging_happy_path(tmp_path, monkeypatch):
    """
    Scenario: one valid skill in staging, not yet in master.
    Expected: skill copied to master, removed from staging, added to results["sync"]["added"].
    """
    staging_dir, master_dir, _ = patch_config(tmp_path, monkeypatch)
    make_skill(staging_dir, "my-new-skill")

    results = m.run_sync(dry_run=False)

    # Promoted to master
    assert (master_dir / "my-new-skill" / "SKILL.md").exists(), \
        "Skill should be promoted to master"
    # Removed from staging
    assert not (staging_dir / "my-new-skill").exists(), \
        "Skill should be removed from staging after promotion"
    # Count incremented
    assert results["sync"]["added"] >= 1, \
        f"Expected added >= 1, got {results['sync']['added']}"
    # Log entry present
    assert any("staged contribution" in log for log in results["sync"]["logs"]), \
        "Expected 'staged contribution' in sync logs"


def test_staging_dry_run(tmp_path, monkeypatch):
    """
    Scenario: dry_run=True — nothing should be written to disk.
    Expected: no copy, no removal, log entry still emitted.
    """
    staging_dir, master_dir, _ = patch_config(tmp_path, monkeypatch)
    make_skill(staging_dir, "dry-skill")

    results = m.run_sync(dry_run=True)

    # NOT copied to master
    assert not (master_dir / "dry-skill").exists(), \
        "dry_run=True should not copy skill to master"
    # NOT removed from staging
    assert (staging_dir / "dry-skill").exists(), \
        "dry_run=True should not remove skill from staging"
    # Count still incremented (would-be additions)
    assert results["sync"]["added"] >= 1, \
        "dry_run should still report would-be additions"


def test_staging_missing_skill_md(tmp_path, monkeypatch):
    """
    Scenario: staging skill dir exists but has no SKILL.md.
    Expected: skipped (not promoted), count stays 0 for staging.
    """
    staging_dir, master_dir, _ = patch_config(tmp_path, monkeypatch)
    make_skill(staging_dir, "incomplete-skill", with_skill_md=False)

    results = m.run_sync(dry_run=False)

    # NOT promoted
    assert not (master_dir / "incomplete-skill").exists(), \
        "Skill without SKILL.md must not be promoted"
    # staging copy still intact
    assert (staging_dir / "incomplete-skill").exists(), \
        "Skill without SKILL.md must remain in staging"
    # No staged contribution log entry
    staged_logs = [l for l in results["sync"]["logs"] if "staged contribution" in l]
    assert len(staged_logs) == 0, \
        "No staged contribution log expected for incomplete skill"


def test_staging_duplicate_skip(tmp_path, monkeypatch):
    """
    Scenario: skill already exists in master (either real dir or symlink).
    Expected: skipped silently, no overwrite, count not incremented for staging.
    """
    staging_dir, master_dir, _ = patch_config(tmp_path, monkeypatch)
    # Pre-populate master with the same skill
    make_skill(master_dir, "existing-skill")
    make_skill(staging_dir, "existing-skill")

    before_added = 0
    results = m.run_sync(dry_run=False)

    # Master copy unchanged (still only one SKILL.md)
    assert (master_dir / "existing-skill" / "SKILL.md").exists()
    # No staged contribution log for this skill
    staged_logs = [l for l in results["sync"]["logs"]
                   if "staged contribution" in l and "existing-skill" in l]
    assert len(staged_logs) == 0, \
        "Duplicate skill in master should not produce a staged contribution log"


def test_staging_dir_absent(tmp_path, monkeypatch):
    """
    Scenario: STAGING_DIR does not exist on disk.
    Expected: run_sync() completes without error, 0 staging additions.
    """
    staging_dir, master_dir, _ = patch_config(tmp_path, monkeypatch)
    # Remove the staging dir that patch_config created
    shutil.rmtree(staging_dir)
    # Override STAGING_DIR to point to the now-absent path
    monkeypatch.setattr(m, "STAGING_DIR", staging_dir)

    # Should not raise
    results = m.run_sync(dry_run=False)

    staged_logs = [l for l in results["sync"]["logs"] if "staged contribution" in l]
    assert len(staged_logs) == 0, \
        "No staging logs expected when STAGING_DIR is absent"


def test_staging_jsonl_event(tmp_path, monkeypatch):
    """
    Scenario: main() with --export-jsonl; a valid staged skill is promoted.
    Expected: CONTRIBUTIONS_LOG contains one line with action='contributed', contributor='gwen'.
    """
    staging_dir, master_dir, contributions_log = patch_config(tmp_path, monkeypatch)
    make_skill(staging_dir, "gwen-skill-alpha")

    # Call main() with --export-jsonl flag by monkeypatching sys.argv
    monkeypatch.setattr(sys, "argv", ["skill-sync.py", "--both", "--export-jsonl"])
    m.main()

    assert contributions_log.exists(), "CONTRIBUTIONS_LOG must be created"
    lines = [json.loads(l) for l in contributions_log.read_text().splitlines() if l.strip()]
    gwen_events = [e for e in lines if e.get("contributor") == "gwen"]
    assert len(gwen_events) >= 1, \
        f"Expected at least 1 gwen contribution event, got: {lines}"
    assert gwen_events[0]["action"] == "contributed", \
        f"Expected action='contributed', got '{gwen_events[0]['action']}'"
    assert gwen_events[0]["skill"] == "gwen-skill-alpha", \
        f"Expected skill='gwen-skill-alpha', got '{gwen_events[0]['skill']}'"
```
